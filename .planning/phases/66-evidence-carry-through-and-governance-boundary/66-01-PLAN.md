```elixir
@audit_summary_keys ~w(
  preview_basis
  preview_schema_version
  preview_fingerprint
  uncertainty
  sample_evidence
  impression_evidence
  affected_reference_keys
)

@spec audit_evidence_summary(map()) :: map()
def audit_evidence_summary(preview) when is_map(preview) do
  preview = normalize_preview_map(preview)

  base = %{
    "preview_basis" => fetch_string(preview, :preview_basis),
    "preview_schema_version" => fetch(preview, :preview_schema_version),
    "preview_fingerprint" => fetch_string(preview, :preview_fingerprint),
    "uncertainty" => normalize_uncertainty(fetch(preview, :uncertainty)),
    "sample_evidence" => normalize_evidence_list(fetch(preview, :sample_evidence)),
    "impression_evidence" => normalize_impression_evidence(fetch(preview, :impression_evidence)),
    "affected_reference_keys" => affected_reference_keys(fetch(preview, :affected_references) || [])
  }

  Enum.reduce(@audit_summary_keys, %{}, fn key, acc ->
    case Map.get(base, key) do
      nil -> acc
      [] -> acc
      %{} = map when map_size(map) == 0 -> acc
      value -> Map.put(acc, key, value)
    end
  end)
end
```

Implement private helpers (or inline):
- `normalize_preview_map/1` — accept atom or string keys (mirror existing `fetch/2` patterns in module)
- `normalize_uncertainty/1` — ensure string keys; preserve `authoritative_population_count?` as false if missing
- `normalize_evidence_list/1` — list of maps with string keys (reuse `normalize_term/1` or AuditEvent-style normalization)
- `normalize_impression_evidence/1` — map with string keys from allowlisted impression fields only
- `fetch_string/2` — nil-safe string coercion

Do **not** include `:affected_references`, `:findings`, `:environment_scope`, or raw `:samples` / `:impression_summary` in output.
